% % Title page
\renewcommand{\maketitle}{ 
	\singlespacing
	\thispagestyle{empty} 
  % \newgeometry{top=2in,left=1in,right=1in,bottom=1in}
	\begin{center}
	  \vspace*{0.8125in}
	  \Large{{\MakeUppercase{$title$}}} \\ \vspace{1in} 
	  \normalsize 
  	$author$ \\ \vspace{1in} 
  	A $doctype$ submitted to the faculty of the University of North Carolina at Chapel Hill in partial fulfillment of the requirements for the degree of $degree$ in the $deprtment$ in the $school$.
  	\vspace{1in}
  	\\ Chapel Hill \\ $date$ \\ 
	\end{center}
	\vspace{1in}
	\hfill
  \begin{minipage}{1.5in} 
      \raggedright
      \doublespacing
      Approved by:\\
      $for(committee)$
        $committee$ \\
      $endfor$
  \end{minipage}
  % \restoregeometry
}

% Blank pages
\newcommand*{\blankpage}{%
\vspace*{\fill}
{\centering Page intentionally left blank. \par}
\vspace{\fill}}
\makeatletter
\renewcommand*{\cleardoublepage}{\clearpage\if@twoside \ifodd\c@page\else
\blankpage
\thispagestyle{empty}
\newpage
\if@twocolumn\hbox{}\newpage\fi\fi\fi}
\makeatother

% Center chapter titles, omit page numbers.
\titleformat{\chapter}[block]%
{\centering\bfseries\Large}%
{\MakeUppercase{\chaptertitlename} \thechapter}%
{0pt}%
{\huge\singlespacing}%
[\thispagestyle{empty}]
\titlespacing*{\chapter}{0pt}{0.6875in}{40pt}

% We want numbered subsubsections
\setcounter{secnumdepth}{3}
\setcounter{tocdepth}{3}
% We don't want crazy vertical spacing.
\raggedbottom
% We don't want abandoned words.
\clubpenalty=10000
\widowpenalty=10000

% Set lengths
\setlength{\topskip}{0pt}
\setlength{\parindent}{4ex}

\newcommand{\mytwoin}[1]{
\vspace*{0.6875in}
\begin{center}
\bfseries\Large\MakeUppercase{#1}
\end{center}
}

\newcommand{\myonein}[1]{
\begin{center}
\bfseries\Large\MakeUppercase{#1}
\end{center}
}

\begin{document}

\frontmatter

$if(title)$
  \maketitle
$endif$

\pagestyle{plain}

$if(copyright)$
  \cleardoublepage
  \thispagestyle{empty}

    $author$ \\
    ALL RIGHTS RESERVED
  \end{center}
$endif$

$if(abstract)$
  \cleardoublepage
  \mytwoin{abstract}
  \begin{center}
    \singlespacing

  \end{center}

$endif$

$if(dedication)$
  \cleardoublepage
  \vspace*{0.75in}

  \end{center}

$for(abbreviations)$
  \item[$abbreviations.abv$] $abbreviations.def$
$endfor$
\end{description}

\mainmatter

$body$

\backmatter

\end{document}
